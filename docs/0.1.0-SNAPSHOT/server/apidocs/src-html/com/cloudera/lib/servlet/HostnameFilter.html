015    package com.cloudera.lib.servlet;
016    
017    
018    import javax.servlet.Filter;
019    import javax.servlet.FilterChain;
020    import javax.servlet.FilterConfig;
021    import javax.servlet.ServletException;
022    import javax.servlet.ServletRequest;
023    import javax.servlet.ServletResponse;
024    import java.io.IOException;
025    import java.net.InetAddress;
026    
027    /**
028     * Filter that resolves the requester hostname.
029     */
030    public class HostnameFilter implements Filter {
031      static final ThreadLocal<String> HOSTNAME_TL = new ThreadLocal<String>();
032    
033      /**
034       * Initializes the filter.
035       * <p/>
036       * This implementation is a NOP.
037       *
038       * @param config filter configuration.
039       * @throws ServletException thrown if the filter could not be initialized.
040       */
041      @Override
042      public void init(FilterConfig config) throws ServletException {
043      }
044    
045      /**
046       * Resolves the requester hostname and delegates the request to the chain.
047       * <p/>
048       * The requester hostname is available via the {@link #get} method.
049       * 
050       * @param request servlet request.
051       * @param response servlet response.
052       * @param chain filter chain.
053       * @throws IOException thrown if an IO error occurrs.
054       * @throws ServletException thrown if a servet error occurrs.
055       */
056      @Override
057      public void doFilter(ServletRequest request, ServletResponse response, FilterChain chain)
058        throws IOException, ServletException {
059        try {
060          String hostname = InetAddress.getByName(request.getRemoteAddr()).getCanonicalHostName();
061          HOSTNAME_TL.set(hostname);
062          chain.doFilter(request, response);
063        }
064        finally {
065          HOSTNAME_TL.remove();
066        }
067      }
068    
069      /**
070       * Returns the requester hostname.
071       * @return  the requester hostname.
072       */
073      public static String get() {
074        return HOSTNAME_TL.get();
075      }
076    
077      /**
078       * Destroys the filter.
079       * <p/>
080       * This implementation is a NOP.
081       */
082      @Override
083      public void destroy() {
084      }
085    }
